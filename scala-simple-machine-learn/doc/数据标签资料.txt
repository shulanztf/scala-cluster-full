-- 积分的记录 LinkPointsRecord link_points_record
SELECT t1.id,t1.created_at, t.created_at,t.* 
from link_points_record t 
LEFT JOIN  link_points_record  t1 on t1.id >= t.id 
                          and t1.created_at >= t.created_at
                          and t1.created_at < date_add(t.created_at, interval 10 second)
where 1=1
and t.member_id = 136654320139
ORDER BY t.id DESC
LIMIT 100
;
